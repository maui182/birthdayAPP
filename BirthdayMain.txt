using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleApp1
{
    class Program
    {
        static void Main(string[] args)
        {
            string DateBirthday = Convert.ToString(DateTime.Today);
           
            Console.Write("Please write your Data Birthday (DD/MM/YYYY):");
            string DateToday = Console.ReadLine().ToString();
            int age = DateTime.Parse(DateBirthday).Year - (DateTime.Parse(DateToday)).Year;
            Console.WriteLine("User's age is: " +age);
            Console.Read();

            if (DateToday.Substring(0, 5) == DateBirthday.Substring(0,5))
            {
                Console.WriteLine("Is it the user's birthday today?");
                Console.Read();

                if (age < 40)
                {
                    Console.WriteLine("Congrats on your Birthday");
                    Console.Read();
                }
                   
                else
                {
                    Console.WriteLine("Sorry, it's your bIrthday");
                    Console.Read();
                }
            } 
        }
    }
}
